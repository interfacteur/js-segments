Interprétation HTML d'un objet js littéral simple via un générateur

sur Githubmerci à William

Côté JavaScript

	const objectToDl =
	{
		*[Symbol.iterator] (o)
		{
			yield '<dl>';
			for (let val in o)
			{
				let dd = typeof o[val] === 'object' ?
					Array.from(objectToDl[Symbol.iterator](o[val])).join('')
					: `'${o[val]}'<span>,</span>`;
				yield  `<dt>${val}</dt><dd>${dd}</dd>`;
			}
			yield '</dl>';
		}
	}
