Interprétation HTML d'un objet js littéral simple

merci à William

Côté JavaScript

	function objectToDl(o) {
		let objectKeys = Object.keys(o);
		return '<dl>'
			+ objectKeys.map(key => {
				let value = typeof o[key] === 'object' ?
					objectToDl(o[key]) : `'${o[key]}'<span>,</span>`;
				return `<dt>${key}</dt><dd>${value}</dd>`;
			})
			.join('')
			+ '</dl>';
	}
